## 通用惰性单例 - 单一职责原则

```javascript
// 管理单例职责
var getSingleton = function(fn){
  let result;
  return function(){
    return result || (result = fn.apply(this, arguments))
  }
}
// 创建实例对象职责
var CreatLayer = function(){
  let div = document.createElement('div');
  div.innerHTML = 'Login';
  div.style.display = 'none';
  document.body.appendChild(div);
  return div
}
```
